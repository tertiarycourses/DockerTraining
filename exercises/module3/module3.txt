# Basic Docker Training
# Module 4: Docker Networking

# List networks
docker network ls 

# Port Binding
# Open 1st terminal (container)
docker run --rm -it -p 1234:1234 ubuntu:14.04 
nc -lp 1234
# Open 2nd terminal (host)
nc localhost 1234
sending data

# Ex: Port Binding
docker run --rm -it -p 1234:1234 -p 4567:4567 -it ubuntu:14.04
nc -lp 1234 | nc -lp 4567

#Open 2nd terminal
nc localhost 1234

#Open 3rd termial
nc localhost 4567

# Ex: Pull Tensorflow image
docker pull tensorflow/tensorflow
docker run --rm -it -p 8888:8888 tensorflow/tensorflow

# Sharing Data from localhost 
docker run --rm -it -v /example/test ubuntu:14.04 bash

Ex: Python 
docker run --rm -it -v "$PWD":/example2/test.py -w /example2 python:3 python test.py

# Sharing Data between containers
docker run --rm -it --volumes-from [containerID] ubuntu:14.04 bash
